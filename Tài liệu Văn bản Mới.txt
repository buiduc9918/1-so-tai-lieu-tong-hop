#include "stm32f10x.h"
#include "stm32f10x_gpio.h"
#include "stm32f10x_rcc.h"
void Delay(unsigned int time);
GPIO_InitTypeDef GPIO_Struct;
int main(void){
SystemInit();
SystemCoreClockUpdate();
// cap clock cho GPIOA
RCC_APB2PeriphClockCmd(RCC_APB2Periph_GPIOC, ENABLE);
//cai dat cho chan PC13
GPIO_Struct.GPIO_Mode = GPIO_Mode_Out_PP;
GPIO_Struct.GPIO_Pin = GPIO_Pin_13;
GPIO_Struct.GPIO_Speed = GPIO_Speed_50MHz;
GPIO_Init(GPIOC, &GPIO_Struct);
while(1){
GPIO_SetBits(GPIOC, GPIO_Pin_13);
Delay(1000);
GPIO_ResetBits(GPIOC, GPIO_Pin_13);
Delay(1000);
}
}
void Delay(unsigned int time){
unsigned int i,j;
for(i=0;i<time;i++){
for(j=0;j< 0x2AFF; j++);
}
}

#include "stm32f10x.h"
#include "stm32f10x_gpio.h"
void GPIO_Configure()
{
	RCC->APB2ENR |= 0x000000010;//Cap clock cho portC hoat dong 
	GPIOC->CRL   &= 0xFFFFFFFF0;
	/*Cau hinh cho toc do vÃ  pin */
	GPIOC->CRL   |= 0x000000003;/*Output push pull*/
}
void delay(int time)
{
	for(int i = 0; i< time; i++)
	{
		for(int j = 0; j < 0x2AFF; j++);//1ms
	}
}
int main(){
	GPIO_Configure();
	while(1){
		GPIOC-> ODR |= 0x000000001;
		delay(100);
		GPIOC->ODR  &=~(0x000000001);
		delay(100);
	}
}
